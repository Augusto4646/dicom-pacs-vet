        /* CONTADOR */

        quill.on('text-change',function(){

        let texto = quill.getText();
        let palavras = texto.trim().split(/\s+/).filter(Boolean);

        document.getElementById('contador').innerText =
        palavras.length + " palavras";

        document.getElementById('status').innerText="Editando...";

        });

        function setSize(){

        let size = document.getElementById('fontSize').value;
        quill.format('size',size);

        }

        /* PDF */

     
function gerarPDF() {
    const element = document.querySelector('.laudo');

    html2pdf().from(element).outputPdf('blob').then(function(pdfBlob) {

        // 🔥 DOWNLOAD LOCAL
        const link = document.createElement('a');
        link.href = URL.createObjectURL(pdfBlob);
        link.download = 'laudo.pdf';
        link.click();

        // 🔥 ENVIAR PRO DJANGO
        const reader = new FileReader();
        reader.readAsDataURL(pdfBlob);
        reader.onloadend = function () {

            fetch('/salvar-pdf/{{ exame.id }}/', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/x-www-form-urlencoded',
                },
                body: 'pdf_base64=' + encodeURIComponent(reader.result)
            })
            .then(res => res.json())
            .then(data => console.log('PDF salvo'));
        };
    });
}


        /* IMAGEM */

        document.getElementById('fileInput').onchange=function(){

        let file=this.files[0];
        let reader=new FileReader();

        reader.onload=function(e){

        let range=quill.getSelection(true);
        quill.insertEmbed(range.index,'image',e.target.result);

        };

        reader.readAsDataURL(file);

        };

        function abrirImagem(){
        document.getElementById('fileInput').click();
        }

        /* DRAG DROP */

        quill.root.addEventListener('dragover',e=>e.preventDefault());

        quill.root.addEventListener('drop',function(e){

        e.preventDefault();

        let file=e.dataTransfer.files[0];

        if(file && file.type.startsWith('image')){

        let reader=new FileReader();

        reader.onload=function(event){

        let range=quill.getSelection(true);
        quill.insertEmbed(range.index,'image',event.target.result);

        };

        reader.readAsDataURL(file);

        }

        });

        /* COLAR IMAGEM SEM DUPLICAR */

        document.addEventListener('paste', function(e){

        let items = e.clipboardData.items;

        for(let i=0;i<items.length;i++){

        if(items[i].type.indexOf('image')!==-1){

        e.preventDefault();

        let file = items[i].getAsFile();
        let reader = new FileReader();

        reader.onload=function(event){

        let range = quill.getSelection(true);
        quill.insertEmbed(range.index,'image',event.target.result);

        };

        reader.readAsDataURL(file);

        }

        }

        });

        /* BOTÕES */

        function copiarHTML(){
        navigator.clipboard.writeText(quill.root.innerHTML);
        }

        function copiarTexto(){
        navigator.clipboard.writeText(quill.getText());
        }

        function limpar(){
        quill.setText('');
        document.getElementById('status').innerText='Não salvo';
        }

        function modelo(){

        quill.root.innerHTML=`

function salvar(){

let html = quill.root.innerHTML;

fetch('/salvar-laudo/',{

method:'POST',

headers:{'Content-Type':'application/json'},

body:JSON.stringify({
laudoHTML:html,
studyInstanceUid:'{{ exame.study_instance_uid }}'
})

})
.then(r => r.json())
.then(data => {

document.getElementById('status').innerText="Salvo";

alert(
"✅ Laudo salvo!\n\n" +
"Código: " + data.codigo + "\n\n" +
"Link: http://localhost:8000/portal/" + data.codigo);
});

}

        /* AUTOSAVE */

        setInterval(function(){

        if(document.getElementById("status").innerText==="Editando..."){
        salvar();
        }

        },10000);

        /* CTRL+S */

        document.addEventListener("keydown",function(e){

        if(e.ctrlKey && e.key==="s"){

        e.preventDefault();
        salvar();

        }

        });

        document.addEventListener('mouseup',()=>resizing=false);
